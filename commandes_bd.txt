=== POSTGRESQL ===

# Lister les tables de chaque base
docker exec -it postgres-db psql -U postgres -d auth_service_db -c "\dt"
docker exec -it postgres-db psql -U postgres -d quiz_db -c "\dt"
docker exec -it postgres-db psql -U postgres -d keycloak_db -c "\dt"

# Voir le contenu des tables - auth_service_db
docker exec -it postgres-db psql -U postgres -d auth_service_db -c "SELECT * FROM users;"
docker exec -it postgres-db psql -U postgres -d auth_service_db -c "SELECT * FROM _prisma_migrations;"

# Voir le contenu des tables - quiz_db
docker exec -it postgres-db psql -U postgres -d quiz_db -c "SELECT * FROM quizzes;"
docker exec -it postgres-db psql -U postgres -d quiz_db -c "SELECT * FROM questions;"
